#include <stdio.h>
int main() 
{
    int n;
    printf("Enter the number of elements in the list: ");
    scanf("%d", &n);
    int list[n];
    printf("Enter %d elements:\n", n);
    for (int i = 0; i < n; i++) 
	{
        scanf("%d", &list[i]);
    }
    int min = list[0];
    int max = list[0];
    for (int i = 1; i < n; i++) 
	{
        if (list[i] < min) 
		{
            min = list[i];
        }
        if (list[i] > max) 
		{
            max = list[i];
        }
    }
    printf("Minimum value in the list: %d\n", min);
    printf("Maximum value in the list: %d\n", max);
    return 0;
}
